import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.List;
import java.util.StringTokenizer;

 
public class Main {
	static int N;
	static int[][] arr;
	static boolean[] c;
	static int min = Integer.MAX_VALUE;
	static int res = 0;
	static List<Integer> team1;
	static List<Integer> team2;
	static int team1_sum;
	static int team2_sum;
	
	public static void main(String[] args) throws IOException{
		StringTokenizer st = null;
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		
		// ì‚¬ëŒ ìˆ˜
		N = Integer.parseInt(br.readLine());
		
		arr = new int[N][N];
		
		// ë½‘ìŒ ìœ ë¬´
		c = new boolean[N];
		
		// ì…ë ¥
		for(int i=0; i<N; i++) {
			st = new StringTokenizer(br.readLine(), " ");
			for(int j=0; j<N; j++) {
				arr[i][j] = Integer.parseInt(st.nextToken());
			}
		}
		
		team1 = new ArrayList<Integer>();
		team2 = new ArrayList<Integer>();
		
		comb(0,0);
		System.out.println(min);
		
	}

	// ì¡°í•©
	static void comb(int idx, int start) {
		
		if(idx == N/2) {			
			for(int i=0; i<N; i++) {
				if(c[i]) {
					team1.add(i);
				}
				else {
					team2.add(i);
				}
			}
			
			// team1 í•©êµ¬í•˜ê¸°
			for(int i=0; i<team1.size(); i++) {
				for(int j=0; j<team1.size(); j++) {
					team1_sum += arr[team1.get(i)][team1.get(j)];
				}
			}
			
			// team2 í•©êµ¬í•˜ê¸°
			for(int i=0; i<team2.size(); i++) {
				for(int j=0; j<team2.size(); j++) {
					team2_sum += arr[team2.get(i)][team2.get(j)];
				}
			}
			
			// ìµœì†Ÿê°’ êµ¬í•˜ê¸°
			min = Math.min(min, Math.abs(team1_sum - team2_sum));
			// ë‘˜ë‹¤ ì´ˆê¸°í™”
			team1.clear();
			team2.clear();
			team1_sum = 0;
			team2_sum = 0;
			
			return;
		}
		
		for(int i=start;i<N; i++) {
			c[i] = true;
			comb(idx+1,i+1);
			c[i] = false;
		}
	}
}
