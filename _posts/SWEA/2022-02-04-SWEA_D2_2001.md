import java.util.Scanner;

public class Solution {
	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int T = sc.nextInt();

		for (int tc = 1; tc <= T; tc++) {
			int N = sc.nextInt();
			int M = sc.nextInt();

			// ì…ë ¥
			int[][] arr = new int[N][N];

			for (int i = 0; i < N; i++) {
				for (int j = 0; j < N; j++) {
					arr[i][j] = sc.nextInt();
				}
			}

			int max = 0;

			// íƒìƒ‰
			for (int i = 0; i < N; i++) {
				for (int j = 0; j < N; j++) {
					
					int sum = 0;
					
					// M*M íŒŒë¦¬ì±„ íƒìƒ‰
					for (int a = 0; a < M; a++) {
						for (int b = 0; b < M; b++) {

							int nx, ny;
							nx = j + b;
							ny = i + a;

							// ë²”ìœ„ ë„˜ì–´ê°€ë©´ ì¤‘ì§€
							if (nx < 0 || nx >= N || ny < 0 || ny >= N) {
								break;
							} else {
								sum += arr[ny][nx];
							}
						}
					}
					
					// ìµœëŒ“ê°’ êµ¬í•˜ê¸°
					max = Math.max(max, sum);
				}
			}

			// ì¶œë ¥
			System.out.println("#" + tc + " " + max);

		}
	}
}
