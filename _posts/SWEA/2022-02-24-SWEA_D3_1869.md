public class Solution {

            int K = Integer.parseInt(st.nextToken());

            st = new StringTokenizer(br.readLine(), " ");

                // ê±¸ë¦¬ëŠ” ì‹œê°„
                list.add(Integer.parseInt(st.nextToken()));

            Collections.sort(list);

            // iëŠ” ì‹œê°„ì„ ì˜ë¯¸
            for(int i = 1; i<=list.get(list.size()-1); i++) {

            }

        }

    }
