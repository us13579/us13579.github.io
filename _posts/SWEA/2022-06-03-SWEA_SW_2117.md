import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.StringTokenizer;

public class Solution {
	static int N, M;

	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st;
		StringBuilder sb = new StringBuilder();

		int tc = Integer.parseInt(br.readLine());

		for (int TC = 1; TC <= tc; TC++) {
			sb.append("#").append(TC).append(" ");

			st = new StringTokenizer(br.readLine(), " ");

			// mapì˜ í¬ê¸°
			N = Integer.parseInt(st.nextToken());

			// ì§‘ì˜ ì§€ë¶ˆë¹„ìš©
			M = Integer.parseInt(st.nextToken());

			ArrayList<Integer> HX = new ArrayList<>();
			ArrayList<Integer> HY = new ArrayList<>();

			// ì§‘ ì…ë ¥
			for (int i = 0; i < N; i++) {
				st = new StringTokenizer(br.readLine(), " ");
				for (int j = 0; j < N; j++) {
					int temp = Integer.parseInt(st.nextToken());
					// ì§‘ì¸ ê²½ìš° ì¢Œí‘œ ì €ì¥
					if (temp == 1) {
						HX.add(i);
						HY.add(j);
					}
				}
			}

			// mapì•ˆì— ìˆëŠ” ì§‘ ê°œìˆ˜
			int H = HX.size();
			// ìµœëŒ€ ì§‘ìˆ˜
			int max = 0;
			// ì¤„ì—¬ê°€ë©´ì„œ êµ¬í•˜ê¸°
			for (int k = (int) Math.sqrt(N * N) + 1; k > 0; k--) {
				int cost = k*k + (k-1)*(k-1);
				for(int X=0; X<N; X++) {
					for(int Y=0; Y<N; Y++) {
						// í¬í•¨ë˜ëŠ” ì§‘ìˆ˜
						int cnt = 0;
						// ê±°ë¦¬ì•ˆì— ì†í•˜ë©´ ì§‘ ê°œìˆ˜ ++
						for(int i=0; i<H; i++) {
							if(Math.abs(HX.get(i)-X) + Math.abs(HY.get(i) - Y)<k) {
								cnt++;
							}
						}
						// ì¡°ê±´ ì¶©ì¡± í•œë‹¤ë©´ ìµœëŒ€ì§‘ ê°œìˆ˜ ê°±ì‹ 
						if(cnt*M >= cost) {
							max = Math.max(max, cnt);
						}
					}
				}
			}
			sb.append(max).append("\n");
		}
		System.out.println(sb);
	}
}
