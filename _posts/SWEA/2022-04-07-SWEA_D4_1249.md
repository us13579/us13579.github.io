import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Comparator;
import java.util.LinkedList;
import java.util.PriorityQueue;
import java.util.Queue;

public class Solution {
	static int N;
	static int[][] map;
	static int[] dr = { -1, 1, 0, 0 };
	static int[] dc = { 0, 0, -1, 1 };
	static int minCost;
	static int INF = Integer.MAX_VALUE;

	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

		int T = Integer.parseInt(br.readLine());

			N = Integer.parseInt(br.readLine());

			map = new int[N][N];

			for (int i = 0; i < N; i++) {
				String str = br.readLine();
				String[] sstr = str.split("");
				for (int j = 0; j < N; j++) {
					map[i][j] = Integer.parseInt(sstr[j]);

	}

	static int dijkstra(int startR, int startC) {
        // ë°©ë¬¸ì²˜ë¦¬ ë°°ì—´
		boolean[][] visited = new boolean[N][N];
		// ì¶œë°œì§€ì—ì„œ ìì‹ ê¹Œì§€ì˜ ìµœì†Œë³µêµ¬ì‹œê°„
		int[][] minTime = new int[N][N];

        // ëª¨ë“  ë°°ì—´ê°’ì„ max ê°’ìœ¼ë¡œ ì„¤ì • ( ì´ˆê¸°í™” )
		for (int i = 0; i < N; i++) {

			}
		}

		// int[] = ì •ì ì •ë³´, ì¶œë°œì§€ì—ì„œ ìì‹ ê¹Œì§€ì˜ ë¹„ìš© r,c,cost
		PriorityQueue<int[]> pQueue = new PriorityQueue<int[]>(new Comparator<int[]>() {

			// cost ë¹„êµ
			@Override
			public int compare(int[] o1, int[] o2) {
				return o1[2] - o2[2];
			}
		});

		minTime[startR][startC] = 0;
		pQueue.offer(new int[] { startR, startC, minTime[startR][startC] });

        // BFS
		int r, c, minCost, nr, nc, current[];
		while (true) {
            // ë„ì°©ì§€ë¼ë©´ ëë‚´ê¸°
			if (r == N - 1 && c == N - 1)
				return minCost;


			// pQueue ì•ˆì˜ ì •ì  ì¤‘ ì¶œë°œì§€ì—ì„œ ìì‹ ìœ¼ë¡œì˜ ë¹„ìš©ì´ ìµœì†Œì¸ ì •ì ì˜ ì •ë³´
			current = pQueue.poll();
			r = current[0];
			c = current[1];
			minCost = current[2];

            // ë°©ë¬¸í•œ ê³³ì´ë¼ë©´ í†µê³¼
			if (visited[r][c])
				continue;

            // ë°©ë¬¸ì²˜ë¦¬
			visited[r][c] = true;

			for (int i = 0; i < 4; i++) {
				nr = r + dr[i];
				nc = c + dc[i];
				if (nr >= 0 && nr < N && nc >= 0 && nc < N && minTime[nr][nc] > minCost + map[nr][nc]) {
					minTime[nr][nc] = minCost + map[nr][nc];
					pQueue.offer(new int[] { nr, nc, minTime[nr][nc] });
				}
			}
		}
	}
}
