	static boolean[] isChecked;

		int T = Integer.parseInt(br.readLine());

			// ê·œì˜ì´ ì¹´ë“œ
			arr = new int[card];
			// ì¸ì˜ì´ ì¹´ë“œ
			arr_in = new int[card];
			isChecked = new boolean[19];
			
			// ìµœì¢… ê·œì˜ì´ ìŠ¹ íŒ¨ 
			win = 0;
			lose = 0;
			
			// í…ŒìŠ¤íŠ¸ ì¼€ì´ìŠ¤ ë§ˆë‹¤ ì…ë ¥

			int index = 0;

	static void permutation(int index) {
		// ê¸°ì €ì¡°ê±´
		if (index == arr.length) {

			// ê·œì˜ì´, ì¸ì˜ì´ ì ìˆ˜ í•©ê³„
			int gyu_sum = 0;

		}

				// ì—­íŠ¸ë˜í‚¹

		}
